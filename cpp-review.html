#1学生成绩管理系统OOP简单

📖 题目描述

设计一个 Student 类，包含私有成员 name（string）、id（int）、score（double）。提供构造函数、print() const 成员函数（输出格式：name id score，score 保留一位小数）、以及 getter 方法。读入 N 个学生，按成绩从高到低排序；成绩相同时按 id 从小到大排序，输出排序结果。

📥 输入格式

第一行一个整数 N（1≤N≤100）。接下来 N 行，每行一个字符串 name（不含空格，长度≤20）、一个整数 id、一个浮点数 score。

📤 输出格式

N 行，每行按 name id score 格式输出，score 保留 1 位小数。

📥 样例输入
3
Alice 1001 95.5
Bob 1002 87.0
Charlie 1003 95.5
📤 样例输出
Alice 1001 95.5
Charlie 1003 95.5
Bob 1002 87.0
#include <iostream>
#include <vector>
#include <algorithm>
#include <iomanip>
using namespace std;

class Student {
    string name; int id; double score;
public:
    Student(string n, int i, double s) : name(n), id(i), score(s) {}
    void print() const {
        cout << name << " " << id << " " << fixed << setprecision(1) << score << endl;
    }
    double getScore() const { return score; }
    int getId() const { return id; }
};

int main() {
    int N; cin >> N;
    vector<Student> v;
    for (int i = 0; i < N; ++i) {
        string name; int id; double score;
        cin >> name >> id >> score;
        v.emplace_back(name, id, score);
    }
    sort(v.begin(), v.end(), [](const Student& a, const Student& b) {
        if (a.getScore() != b.getScore()) return a.getScore() > b.getScore();
        return a.getId() < b.getId();
    });
    for (const auto& s : v) s.print();
}

📖 题目描述

实现模板类 MyVector<T>，三指针 _start/_finish/_endOfStorage。必须实现：push_back（二倍扩容）、pop_back、size()、capacity()、operator[]、at(size_t)（越界抛 out_of_range）、insert(size_t pos, const T&)、erase(size_t pos)、析构（逐个析构元素）。输入一系列指令模拟操作。

📥 输入格式

每行一条指令：pb x（push_back）、pop、at i（输出第 i 个）、size、cap、ins i x（在 i 位置插入 x）、era i（删除第 i 个）、end 结束。

📥 样例输入

template<typename T>
class MyVector {
    T* _start=nullptr,*_finish=nullptr,*_endOfStorage=nullptr;
public:
    size_t size()const{return _finish-_start;}
    size_t capacity()const{return _endOfStorage-_start;}

    void reserve(size_t n){
        if(n<=capacity())return;
        T* nd=static_cast<T*>(operator new(n*sizeof(T)));
        size_t sz=size();
        for(size_t i=0;i<sz;++i)new(nd+i) T(std::move(_start[i]));
        for(size_t i=0;i<sz;++i)_start[i].~T();
        operator delete(_start);
        _start=nd;_finish=_start+sz;_endOfStorage=_start+n;
    }

    void push_back(const T& v){
        if(_finish==_endOfStorage)reserve(capacity()==0?1:capacity()*2);
        new(_finish) T(v);++_finish;
    }

    void pop_back(){if(size()){--_finish;_finish->~T();}}
    T& operator[](size_t i){return _start[i];}

    T& at(size_t i){if(i>=size())throw out_of_range("out of range");return _start[i];}

    void insert(size_t pos,const T& v){
        if(pos>size())throw out_of_range("pos");
        if(_finish==_endOfStorage)reserve(capacity()==0?1:capacity()*2);
        for(T* p=_finish;p>_start+pos;--p)new(p)T(std::move(*(p-1)));
        *(_start+pos)=v;++_finish;
    }

    void erase(size_t pos){
        if(pos>=size())throw out_of_range("pos");
        for(T* p=_start+pos;p<_finish-1;++p)*p=std::move(*(p+1));
        --_finish;_finish->~T();
    }

    ~MyVector(){for(size_t i=0;i<size();++i)_start[i].~T();operator delete(_start);}
};

using namespace std;

template<typename T>
class MyList {
    struct Node{T data;Node*prev,*next;Node(const T& d,Node*p=nullptr,Node*n=nullptr):data(d),prev(p),next(n){}};
    Node* head;size_t sz;
public:
    class iterator{Node*p;public:iterator(Node*n):p(n){}T&operator*(){return p->data;}iterator& operator++(){p=p->next;return*this;}bool operator!=(iterator o){return p!=o.p;}Node*get(){return p;}};

#include <iostream>
#include <string>
using namespace std;

template<typename T>
class MyStack{
    struct Node{T data;Node*next;Node(const T& d,Node*n=nullptr):data(d),next(n){}};
    Node* tp;size_t cnt;
public:
    MyStack():tp(nullptr),cnt(0){}
    void push(const T& v){tp=new Node(v,tp);++cnt;}
    void pop(){if(empty())return;Node*o=tp;tp=tp->next;delete o;--cnt;}
    T& top(){return tp->data;}
    bool empty(){return cnt==0;}
    ~MyStack(){while(!empty())pop();}
};

bool isValid(const string& s){
    MyStack<char> st;

    }
    return st.empty();
}

📖 题目描述

实现循环队列 MyQueue<T>（环形缓冲区，push/pop/front/empty，满时 2 倍扩容）。模拟 CPU 时间片轮转调度：N 个任务各有 id 和 time，时间片 Q。每次取队首，若剩余时间≤Q 则完成并输出完成时刻；否则减 Q 后放回队尾。

📥 样例输入
3 100
p1 150
p2 80
p3 200
📤 样例输出
p2 180
p1 330
p3 530
#include <iostream>
#include <string>
using namespace std;

template<typename T>
class MyQueue{
    T* data;size_t cap,head,tail,cnt;
    void expand(){T* nd=new T[cap*2];for(size_t i=0;i<cnt;++i)nd[i]=data[(head+i)%cap];delete[]data;data=nd;head=0;tail=cnt;cap*=2;}
public:
    MyQueue(size_t c=16):cap(c),head(0),tail(0),cnt(0){data=new T[cap];}
    void push(const T& v){if(cnt==cap)expand();data[tail]=v;tail=(tail+1)%cap;++cnt;}
    void pop(){if(!empty()){head=(head+1)%cap;--cnt;}}
    T& front(){return data[head];}
    bool empty(){return cnt==0;}
    ~MyQueue(){delete[]data;}
};

struct Task{string id;int time;};

int main(){
    int N,Q;cin>>N>>Q;MyQueue<Task> q;
    for(int i=0;i<N;++i){Task t;cin>>t.id>>t.time;q.push(t);}
    int elapsed=0;
    while(!q.empty()){Task t=q.front();q.pop();
        if(t.time<=Q){elapsed+=t.time;cout<<t.id<<" "<<elapsed<<endl;}
        else{elapsed+=Q;t.time-=Q;q.push(t);}

#include <vector>
#include <functional>
using namespace std;

template<typename T,typename Cmp=less<T>>
class MyPriorityQueue{
    vector<T> heap;Cmp cmp;
    void siftUp(size_t i){while(i>0){size_t p=(i-1)/2;if(!cmp(heap[p],heap[i]))break;swap(heap[p],heap[i]);i=p;}}
    void siftDown(size_t i){size_t n=heap.size();while(true){size_t best=i,l=2*i+1,r=2*i+2;if(l<n&&cmp(heap[best],heap[l]))best=l;if(r<n&&cmp(heap[best],heap[r]))best=r;if(best==i)break;swap(heap[i],heap[best]);i=best;}}

    bool empty(){return heap.empty();}
    size_t size(){return heap.size();}
};

#include <iostream>
#include <unordered_map>
#include <string>
using namespace std;

class LRUCache{
    struct Node{int key,val;Node*prev,*next;Node(int k=0,int v=0):key(k),val(v),prev(nullptr),next(nullptr){}};
    unordered_map<int,Node*> mp;Node *head,*tail;int cap,sz;
    void addHead(Node* n){n->next=head->next;n->prev=head;head->next->prev=n;head->next=n;}
    void remove(Node* n){n->prev->next=n->next;n->next->prev=n->prev;}
    void moveHead(Node* n){remove(n);addHead(n);}
    Node* popTail(){Node* n=tail->prev;remove(n);return n;}
public:
    LRUCache(int c):cap(c),sz(0){head=new Node();tail=new Node();head->next=tail;tail->prev=head;}

    void put(int k,int v){if(mp.count(k)){Node* n=mp[k];n->val=v;moveHead(n);return;}Node* n=new Node(k,v);mp[k]=n;addHead(n);++sz;if(sz>cap){Node* t=popTail();mp.erase(t->key);delete t;--sz;}}
};
